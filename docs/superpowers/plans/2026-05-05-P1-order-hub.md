# P1 全渠道订单聚合 Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: superpowers:subagent-driven-development.

**Goal:** Build OrderHub domain service for unified cross-platform order management, real-time inventory sync to prevent overselling across channels, KDS dispatch integration, and order management UI.

**Architecture:** OrderHub wraps OmniChannelService with query/statistics APIs consumed by the admin dashboard. An InventorySyncService listens for INVENTORY.CONSUMED events and calls each platform adapter's update_stock(). KDS dispatch is already triggered by OmniChannelService (reuses existing path).

**Tech Stack:** Python 3.11+, FastAPI, SQLAlchemy 2.0 async, Pydantic V2, structlog, React 18 + TypeScript + Tailwind (frontend)

```python
"""OrderHub — cross-platform unified order query engine.

Provides unified query/statistics/filter across all delivery platforms.
All methods receive tenant_id explicitly for RLS compliance.
"""
from __future__ import annotations

from dataclasses import dataclass
from datetime import datetime
from typing import Optional

import structlog
from sqlalchemy import and_, asc, desc, func, or_, select, text
from sqlalchemy.ext.asyncio import AsyncSession

from .omni_channel_service import (
    OmniChannelService,
    omni_order_match_clause,
    row_omni_platform_key,
    row_omni_platform_order_id,
)

logger = structlog.get_logger()


@dataclass
class OrderHubFilters:
    platform: str = ""
    status: str = ""
    store_id: str = ""
    keyword: str = ""
    date_from: Optional[datetime] = None
    date_to: Optional[datetime] = None
    page: int = 1
    size: int = 20


class OrderHub:
    """Unified order hub for cross-platform order queries."""

    PLATFORMS = OmniChannelService.PLATFORMS

    def __init__(self, db: AsyncSession, tenant_id: str):
        self.db = db
        self.tenant_id = tenant_id

    async def list_orders(self, filters: OrderHubFilters) -> dict:
        conditions = [
            text("o.tenant_id = :tenant_id"),
            text("o.order_type = 'delivery'"),
        ]
        params = {"tenant_id": self.tenant_id}

        if filters.platform and filters.platform in self.PLATFORMS:
            conditions.append(text("o.sales_channel_id = :platform"))
            params["platform"] = f"delivery_{filters.platform}"
        if filters.status:
            conditions.append(text("o.status = :status"))
            params["status"] = filters.status
        if filters.store_id:
            conditions.append(text("o.store_id = :store_id"))
            params["store_id"] = filters.store_id
        if filters.keyword:
            conditions.append(text(
                "(o.order_metadata->>'platform_order_id' ILIKE :kw "
                "OR o.order_metadata->>'customer_phone' ILIKE :kw)"
            ))
            params["kw"] = f"%{filters.keyword}%"
        if filters.date_from:
            conditions.append(text("o.created_at >= :date_from"))
            params["date_from"] = filters.date_from
        if filters.date_to:
            conditions.append(text("o.created_at <= :date_to"))
            params["date_to"] = filters.date_to

        where = " AND ".join(str(c) for c in conditions)
        offset = (filters.page - 1) * filters.size

        count_sql = text(f"SELECT COUNT(*) FROM orders o WHERE {where}")
        total = await self.db.scalar(count_sql, params)

        query_sql = text(f"""
            SELECT o.id, o.tenant_id, o.store_id, o.sales_channel_id,
                   o.status, o.total_fen, o.order_metadata, o.created_at, o.updated_at
            FROM orders o
            WHERE {where}
            ORDER BY o.created_at DESC
            LIMIT :limit OFFSET :offset
        """)
        params["limit"] = filters.size
        params["offset"] = offset
        rows = (await self.db.execute(query_sql, params)).fetchall()

        return {
            "items": [_row_to_order_summary(r) for r in rows],
            "total": total,
            "page": filters.page,
            "size": filters.size,
        }

    async def get_order_detail(self, order_id: str) -> dict | None:
        sql = text("""
            SELECT o.* FROM orders o
            WHERE o.id = :order_id AND o.tenant_id = :tenant_id
              AND o.order_type = 'delivery'
        """)
        row = (await self.db.execute(sql, {
            "order_id": order_id, "tenant_id": self.tenant_id,
        })).first()
        if not row:
            return None

        canonical_sql = text("""
            SELECT * FROM channel_canonical_orders
            WHERE canonical_order_id = :order_id AND tenant_id = :tenant_id
        """)
        canonical = (await self.db.execute(canonical_sql, {
            "order_id": order_id, "tenant_id": self.tenant_id,
        })).first()

        return {
            "order": _row_to_order_detail(row),
            "canonical": dict(canonical._mapping) if canonical else None,
            "platform": row_omni_platform_key(row, self.PLATFORMS),
        }

    async def get_stats(self, store_id: str = "", platform: str = "") -> dict:
        conditions = [
            text("o.tenant_id = :tenant_id"),
            text("o.order_type = 'delivery'"),
        ]
        params = {"tenant_id": self.tenant_id}
        if platform and platform in self.PLATFORMS:
            conditions.append(text("o.sales_channel_id = :platform"))
            params["platform"] = f"delivery_{platform}"
        if store_id:
            conditions.append(text("o.store_id = :store_id"))
            params["store_id"] = store_id

        where = " AND ".join(str(c) for c in conditions)
        sql = text(f"""
            SELECT
                COUNT(*) as total_orders,
                COALESCE(SUM(o.total_fen), 0) as total_fen,
                COUNT(*) FILTER (WHERE o.status = 'pending') as pending,
                COUNT(*) FILTER (WHERE o.status = 'confirmed') as confirmed,
                COUNT(*) FILTER (WHERE o.status = 'completed') as completed,
                COUNT(*) FILTER (WHERE o.status = 'cancelled') as cancelled
            FROM orders o WHERE {where}
        """)
        row = (await self.db.execute(sql, params)).first()
        m = row._mapping
        return {
            "total_orders": m["total_orders"],
            "total_fen": m["total_fen"],
            "pending": m["pending"],
            "confirmed": m["confirmed"],
            "completed": m["completed"],
            "cancelled": m["cancelled"],
        }


def _row_to_order_summary(row: Any) -> dict:
    m = row._mapping
    meta = m.get("order_metadata") or {}
    return {
        "id": str(m["id"]),
        "tenant_id": str(m["tenant_id"]),
        "store_id": str(m["store_id"]),
        "platform": (m.get("sales_channel_id") or "").replace("delivery_", ""),
        "status": m["status"],
        "total_fen": m["total_fen"],
        "platform_order_id": meta.get("platform_order_id", ""),
        "customer_phone": meta.get("customer_phone", ""),
        "created_at": str(m["created_at"]),
    }


def _row_to_order_detail(row: Any) -> dict:
    m = row._mapping
    meta = m.get("order_metadata") or {}
    return {
        "id": str(m["id"]),
        "tenant_id": str(m["tenant_id"]),
        "store_id": str(m["store_id"]),
        "platform": (m.get("sales_channel_id") or "").replace("delivery_", ""),
        "status": m["status"],
        "total_fen": m["total_fen"],
        "order_metadata": meta,
        "created_at": str(m["created_at"]),
        "updated_at": str(m["updated_at"]),
    }
```

"""OrderHub API — cross-platform order queries."""
from __future__ import annotations

from datetime import datetime

from fastapi import APIRouter, Depends, HTTPException, Query, Request
from sqlalchemy.ext.asyncio import AsyncSession

from shared.ontology.src.database import get_db

from ..services.order_hub import OrderHub, OrderHubFilters

router = APIRouter(prefix="/api/v1/orders", tags=["order-hub"])


def _tenant_id(request: Request) -> str:
    tid = getattr(request.state, "tenant_id", None) or request.headers.get("X-Tenant-ID", "")
    if not tid:
        raise HTTPException(status_code=400, detail="X-Tenant-ID required")
    return tid


@router.get("")
async def list_orders(
    request: Request,
    platform: str = Query(""),
    status: str = Query(""),
    store_id: str = Query(""),
    keyword: str = Query(""),
    page: int = Query(1, ge=1),
    size: int = Query(20, ge=1, le=100),
    db: AsyncSession = Depends(get_db),
):
    hub = OrderHub(db, _tenant_id(request))
    filters = OrderHubFilters(
        platform=platform, status=status, store_id=store_id,
        keyword=keyword, page=page, size=size,
    )
    return {"ok": True, "data": await hub.list_orders(filters)}


@router.get("/stats")
async def order_stats(
    request: Request,
    store_id: str = Query(""),
    platform: str = Query(""),
    db: AsyncSession = Depends(get_db),
):
    hub = OrderHub(db, _tenant_id(request))
    return {"ok": True, "data": await hub.get_stats(store_id=store_id, platform=platform)}


@router.get("/{order_id}")
async def order_detail(
    order_id: str, request: Request, db: AsyncSession = Depends(get_db),
):
    hub = OrderHub(db, _tenant_id(request))
    detail = await hub.get_order_detail(order_id)
    if not detail:
        raise HTTPException(status_code=404, detail="Order not found")
    return {"ok": True, "data": detail}
```

"""InventorySyncService — real-time stock sync across delivery platforms.

Listens for INVENTORY.CONSUMED events and pushes stock updates to all
connected platforms via their adapters.
"""
from __future__ import annotations

import asyncio
import os
from typing import Any

import structlog

logger = structlog.get_logger()

PLATFORM_ORDER = ("meituan", "eleme", "douyin", "amap", "taobao")


class InventorySyncService:
    async def sync_stock_to_platform(
        self, sku_id: str, stock: int, platform: str
    ) -> bool:
        from shared.adapters.delivery_factory import get_delivery_adapter

        try:
            adapter = get_delivery_adapter(
                platform,
                config={
                    "app_key": os.environ.get(f"{platform.upper()}_APP_KEY", ""),
                    "app_secret": os.environ.get(f"{platform.upper()}_APP_SECRET", ""),
                },
            )
            result = await adapter.update_stock(sku_id, stock)
            await adapter.close()
            return bool(result)
        except (ValueError, ImportError) as exc:
            logger.warning(
                "inventory_sync.platform_skip",
                platform=platform, sku_id=sku_id, reason=str(exc),
            )
            return False

    async def sync_stock_to_all_platforms(
        self, sku_id: str, stock: int
    ) -> dict[str, bool]:
        tasks = {
            p: self.sync_stock_to_platform(sku_id, stock, p)
            for p in PLATFORM_ORDER
        }
        results = {}
        for p, task in tasks.items():
            try:
                results[p] = await task
            except Exception:
                logger.exception("inventory_sync.platform_failed", platform=p, sku_id=sku_id)
                results[p] = False

        succeeded = sum(1 for v in results.values() if v)
        logger.info(
            "inventory_sync.done",
            sku_id=sku_id, stock=stock,
            succeeded=f"{succeeded}/{len(results)}",
        )
        return results

    async def handle_inventory_consumed(self, event: dict[str, Any]) -> None:
        sku_id = event.get("sku_id", "")
        remaining = event.get("remaining", 0)
        if not sku_id:
            return
        # If stock = 0, push "sold out" to all platforms
        await self.sync_stock_to_all_platforms(sku_id, remaining)
```

"""Inventory sync trigger API."""
from __future__ import annotations

from fastapi import APIRouter, BackgroundTasks, HTTPException, Request

from ..services.inventory_sync_service import InventorySyncService

router = APIRouter(prefix="/api/v1/inventory-sync", tags=["inventory-sync"])


@router.post("/trigger")
async def trigger_stock_sync(
    request: Request,
    sku_id: str,
    stock: int,
    background_tasks: BackgroundTasks,
):
    """Manually trigger stock sync to all platforms."""
    svc = InventorySyncService()
    background_tasks.add_task(svc.sync_stock_to_all_platforms, sku_id, stock)
    return {"ok": True, "data": {"sku_id": sku_id, "stock": stock, "status": "queued"}}
